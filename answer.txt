public int compareTo(Media m) {
	int compareTitle = this.title.compareToIgnoreCase(m.title);
	if (compareTitle < 0) return -1;
	if (compareTitle > 0) return 1;
	if (this.length > m.length) return -1;
	if (this.length < m.length) return 1;
	if (this.cost < m.cost) return -1;
	if (this.cost > m.cost) return 1;
	return 0; }
